accion Sistema de reservas de cine es
    ambiente 
        registro = pelicula
            nombre=:= alfanumerico
            preciobase=:= real
        fin registro

        registro = horario
            peliculaindice=:= entero
            hora=:= alfanumerico
            capacidadtotal=:= entero
            entradasvendidas=:= entero
        fin registro

        // arreglos globales
        peliculas[]:= arreglo de pelicula
        horarios[]:= arreglo de horario

        // acumuladores/contadores para estadísticas

        totalentradasvendidas:= 0
        totalrecaudado:= 0
        ventasporpelicula[]:= arreglo de entero   // paralelo a peliculas[]
        ventasporhorario[]:= arreglo de entero    // paralelo a horarios[]

        // módulo de validación genérica
         funcion esopcionvalida(valor:= entero, min:= entero, max:= entero):= booleano
            si valor >= min y valor <= max entonces
                verdadero
            sino
                falso
            fin si
        fin funcion    

        //selección de película y horario

        procedimiento peliculas()
            para i = 0 hasta longitud(peliculas) - 1 hacer
                esc ( (i+1) + ". " + peliculas[i].nombre + " - $" + peliculas[i].preciobase)
            fin para
        fin procedimiento

        funcion seleccionarpelicula():= entero
            esc (peliculas())
            indice:= entero = leerenterovalidado("seleccione una película:=", 1, longitud(peliculas))
            indice:=indice - 1
        fin funcion

        funcion horariosdisponibles(peliculaidx:= entero):= entero
            cantidadencontrados:= 0
            
            para i = 0 hasta longitud(horarios) - 1 hacer
                si horarios[i].peliculaindice == peliculaidx entonces
                    disponibles:= entero = horarios[i].capacidadtotal - horarios[i].entradasvendidas
                    esc ( (i+1) + ". " + horarios[i].hora + " (disponibles:= " + disponibles + ")")
                    cantidadencontrados := cantidadencontrados + 1
                fin si
            fin para
            
            esc (cantidadencontrados)
        fin funcion

        funcion seleccionarhorario(peliculaidx:= entero):= entero
            cantidad:= horariosdisponibles(peliculaidx)
            
            si cantidad == 0 entonces
                esc ( "error:= no hay horarios disponibles para esta película.")
                cantidad:=cantidad-1
            fin si
            
            indicehorario:=leerenterovalidado("seleccione un horario:=", 1, longitud(horarios))
            indicehorario:= indicehorario - 1
        fin funcion
     

        //control de capacidad, promociones y cálculo de importe

        funcion haycapacidaddisponible(horarioidx:= entero, cantidadsolicitada:= entero):= booleano
            disponibles:= horarios[horarioidx].capacidadtotal - horarios[horarioidx].entradasvendidas
            (cantidadsolicitada <= disponibles)
        fin funcion

        funcion calculardescuentopromocion(cantidad:= entero):= real
            // ejemplo de promoción:= 4 o más entradas -> 15% descuento, 2-3 entradas -> 10%
            si cantidad >= 4 entonces
                calculardescuentopromocion:= 0.15
            sino si cantidad >= 2 entonces
                calculardescuentopromocion:= 0.10
            sino
                calculardescuentopromocion:=  0
            fin si
        fin funcion

        funcion calcularimporte(preciobase:= real, cantidad:= entero):= real
            subtotal:= preciobase * cantidad
            descuento:= calculardescuentopromocion(cantidad)
            total:= subtotal - (subtotal * descuento)
            calcularimporte:=total
        fin funcion
            
        
        // registro de venta y actualización de estadísticas

        procedimiento registrarventa(peliculaidx:= entero, horarioidx:= entero, cantidad:= entero, importe:= real)
            horarios[horarioidx].entradasvendidas := horarios[horarioidx].entradasvendidas + cantidad
            
            // acumuladores globales
            totalentradasvendidas := totalentradasvendidas + cantidad
            totalrecaudado := totalrecaudado + importe
            
            // contadores para estadísticas
            ventasporpelicula[peliculaidx] := ventasporpelicula[peliculaidx] + cantidad
            ventasporhorario[horarioidx] := ventasporhorario[horarioidx] + cantidad
        fin procedimiento    

        
        funcion leerenterovalidado(mensaje:= cadena, min:= entero, max:= entero):= entero
            valor: entero
            valido: booleano 

            valido:= falso
            
            repetir
                esc ( mensaje)
                valor = leer entero
                si no esopcionvalida(valor, min, max) entonces
                    esc ( "error:= debe ingresar un número entre " + min + " y " + max)
                sino
                    valido = verdadero
                fin si
            hasta que valido == verdadero
            
        fin funcion

        //flujo de reserva (integra todo lo anterior)

        procedimiento realizarreserva()
            peliculaidx:= seleccionarpelicula()
            horarioidx:= seleccionarhorario(peliculaidx)
            
            si horarioidx == -1 entonces
                // sale del procedimiento, ya se informó el error
            fin si
            
            cantidad:= entero
            reservavalida:= falso
            
            repetir
                cantidad := leerenterovalidado("ingrese cantidad de entradas (1-10):=", 1, 10)
                
                si no haycapacidaddisponible(horarioidx, cantidad) entonces
                    disponibles:= horarios[horarioidx].capacidadtotal - horarios[horarioidx].entradasvendidas
                    esc ( "error:= no hay suficiente capacidad. solo quedan " + disponibles + " lugares.")
                sino
                    reservavalida := verdadero
                fin si
            hasta que reservavalida == verdadero
            
            preciobase:= peliculas[peliculaidx].preciobase
            importetotal:= calcularimporte(preciobase, cantidad)
            
            registrarventa(peliculaidx, horarioidx, cantidad, importetotal)
            
            esc ( "----- comprobante de reserva -----")
            esc ( "película:= " + peliculas[peliculaidx].nombre)
            esc ( "horario:= " + horarios[horarioidx].hora)
            esc ( "cantidad de entradas:= " + cantidad)
            esc ( "importe total:= $" + importetotal)
            esc ( "¡reserva realizada con éxito!")
        fin procedimiento

        //estadísticas

        procedimiento estadisticas()
            si totalentradasvendidas == 0 entonces
                esc ( "aún no se registraron ventas.")
                retornar
            fin si
            
            // película más elegida
            indicemax:= 0
            para i = 1 hasta longitud(ventasporpelicula) - 1 hacer
                si ventasporpelicula[i] > ventasporpelicula[indicemax] entonces
                    indicemax := i
                fin si
            fin para
            
            // horario con mayor demanda
            indicehorariomax:= 0
            para i = 1 hasta longitud(ventasporhorario) - 1 hacer
                si ventasporhorario[i] > ventasporhorario[indicehorariomax] entonces
                    indicehorariomax := i
                fin si
            fin para
            
            esc ( "===== estadísticas =====")
            esc ( "total de entradas vendidas:= " + totalentradasvendidas)
            esc ( "total recaudado:= $" + totalrecaudado)
            esc ( "película más elegida:= " + peliculas[indicemax].nombre + " (" + ventasporpelicula[indicemax] + " entradas)")
            esc ( "horario con mayor demanda:= " + horarios[indicehorariomax].hora + " (" + ventasporhorario[indicehorariomax] + " entradas)")
        fin procedimiento

        // inicialización de datos (carga de ejemplo)
        procedimiento inicializardatos()
            peliculas[0] = {"interestelar", 1500.0}
            peliculas[1] = {"el padrino", 1200.0}
            peliculas[2] = {"toy story", 1000.0}
            
            horarios[0] = {0, "14:=00", 50, 0}
            horarios[1] = {0, "18:=00", 50, 0}
            horarios[2] = {1, "16:=00", 40, 0}
            horarios[3] = {2, "20:=00", 60, 0}
            
            para i = 0 hasta longitud(peliculas) - 1 hacer
                ventasporpelicula[i] = 0
            fin para
            
            para i = 0 hasta longitud(horarios) - 1 hacer
                ventasporhorario[i] = 0
            fin para
        fin procedimiento
            
    proceso 
        inicializardatos()
        opcion:= entero
        
        repetir
            esc ( "===== sistema de reservas de cine =====")
            esc ( "1. reservar entradas")
            esc ( "2. ver estadísticas")
            esc ( "3. salir")
            opcion = leer entero
            
            si no esopcionvalida(opcion, 1, 3) entonces
                esc ( "error:= opción inválida. ingrese un número entre 1 y 3.")
            sino
                segun opcion hacer
                    caso 1:= realizarreserva()
                    caso 2:= esc (estadisticas())
                    caso 3:= esc ( "gracias por usar el sistema. ¡hasta pronto!")
                fin segun
            fin si
        hasta que opcion == 3
    fin accion
